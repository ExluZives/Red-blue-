        end

            if child:IsA("TextButton") then
                child.BackgroundColor3 = colors.Button
                child.TextColor3 = colors.Text
            elseif child:IsA("Frame") then
                for _, grandChild in ipairs(child:GetChildren()) do
                    if grandChild:IsA("TextLabel") then
                        grandChild.TextColor3 = colors.Text
                    elseif grandChild:IsA("TextButton") then
                        grandChild.BackgroundColor3 = colors.Button
                        grandChild.TextColor3 = colors.Text
                    end
                end
            end
        end
        settingsPanel.BackgroundColor3 = colors.Background
        settingsPanelStroke.Color = colors.Stroke
        Settings.DayMode = (theme == "Light")
        saveSettings()
    end

        btn.MouseButton1Click:Connect(function()
            Settings[settingKey] = not Settings[settingKey]
            btn.Text = name .. ": " .. (Settings[settingKey] and "ON" or "OFF")
            toggleSound:Play()
            if callback then
                callback(Settings[settingKey])

            if settingKey == "DayMode" then
                applyTheme(Settings.DayMode and "Light" or "Dark")
            end
        end)
        return btn
    end

-- Дебаунс для обработки кликов
local function debounce(func, delay)
    local lastCall = 0
    return function(...)
        local now = tick()
        if now - lastCall >= delay then
            lastCall = now
            return func(...)
        end
    end
end

        local closest, dist = nil, Settings.FOV
        local center = Vector2.new(Camera.ViewportSize.X / 2 + Settings.AimOffsetX, Camera.ViewportSize.Y / 2 + Settings.AimOffsetY)
        local maxDistance = 1000
        
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 and p.Character:FindFirstChild("HumanoidRootPart") then
                if Settings.TeamCheck and p.Team == LocalPlayer.Team then continue end
                local part
                if Settings.Part == "Head" then
                    part = p.Character:FindFirstChild("Head")
                elseif Settings.Part == "Torso" then
                    part = p.Character:FindFirstChild("Torso") or p.Character:FindFirstChild("UpperTorso") or p.Character:FindFirstChild("HumanoidRootPart")
                end
                if not part then continue end
                
                local distanceToPlayer = ((part.Name == "Head" and part.Position + Vector3.new(0, 0.3, 0) or part.Position) - Camera.CFrame.Position).Magnitude
                if distanceToPlayer > maxDistance then continue end
                
                local screenPos, onScreen = Camera:WorldToViewportPoint((part.Name == "Head" and part.Position + Vector3.new(0, 0.3, 0) or part.Position))
                if not onScreen then continue end
                local distance = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
                if distance <= Settings.FOV then
                    if Settings.WallCheck and not isVisible(part) then continue end
                    closest = part
                    dist = distance
                end
            end
        end
        return closest
    end

    local function getBulletSpeed()
        local character = LocalPlayer.Character
        if not character then return 300 end

        local tool = character:FindFirstChildOfClass("Tool")
        if not tool then return 300 end

        local possibleNames = {
            "BulletSpeed", "ProjectileSpeed", "Velocity", "Speed",
            "MuzzleVelocity", "FireSpeed", "ProjectileVelocity", "BulletVelocity",
            "ShotSpeed", "LaunchSpeed", "InitialSpeed", "MuzzleSpeed"
        }

        local function recursiveSearch(obj)
            for _, name in ipairs(possibleNames) do
                local val = obj:FindFirstChild(name)
                if val and tonumber(val.Value) and val.Value > 0 then
                    return val.Value
                end
            end
            for _, child in ipairs(obj:GetChildren()) do
                local found = recursiveSearch(child)
                if found then return found end
            end
            return nil
        end

        local config = tool:FindFirstChild("Configuration") or tool:FindFirstChild("Settings") or tool
        local speed = recursiveSearch(config)
        if speed then return speed end

        for _, child in ipairs(tool:GetDescendants()) do
            if child:IsA("ModuleScript") then
                local success, data = pcall(require, child)
                if success and type(data) == "table" then
                    for _, name in ipairs(possibleNames) do
                        if tonumber(data[name]) and data[name] > 0 then
                            return data[name]
                        end
                    end
                end
            end
        end

        local replicated = game:GetService("ReplicatedStorage")
        local repSpeed = recursiveSearch(replicated)
        if repSpeed then return repSpeed end

        return 300
    end

    local function calculatePredictionTime(target, distanceToTarget)
        if not Settings.AutoPredict then return 0 end
        
        local bulletSpeed = getBulletSpeed()
        local targetVelocity = target.Parent:FindFirstChild("HumanoidRootPart") and target.Parent.HumanoidRootPart.Velocity or Vector3.new(0, 0, 0)
        local velocityMagnitude = targetVelocity.Magnitude
        
        local basePrediction = distanceToTarget / bulletSpeed
        local velocityFactor = velocityMagnitude / 100
        local predictionTime = basePrediction * (1 + velocityFactor * 0.5)
        
        return math.clamp(predictionTime, 0.05, 0.3)
    end

    local function getAimDotPosition()
        local degPerPixelX = Camera.FieldOfView / Camera.ViewportSize.Y
        local degPerPixelY = (Camera.FieldOfView * (Camera.ViewportSize.Y / Camera.ViewportSize.X)) / Camera.ViewportSize.Y

        local offsetXAngle = math.rad(Settings.AimOffsetX * degPerPixelX)
        local offsetYAngle = math.rad(Settings.AimOffsetY * degPerPixelY)

        local targetWorld = (Camera.CFrame * CFrame.Angles(-offsetYAngle, -offsetXAngle, 0)).LookVector * 1000
        local worldPos = Camera.CFrame.Position + targetWorld

        local screenPos, onScreen = Camera:WorldToViewportPoint(worldPos)
        if onScreen then
            return Vector2.new(screenPos.X, screenPos.Y)
        else
            return getAimingCenter()
        end
    end

    local lastUpdate = 0

    local raycastInterval = 0.1
    local lastTarget = nil
    local currentTargetPart = Settings.Part
    local lastTargetValid = false
    local lastTargetPos = nil

    RunService.RenderStepped:Connect(function(deltaTime)
        if Settings.Part ~= currentTargetPart then
            lastTarget = nil
            lastTargetValid = false
            lastTargetPos = nil
            currentTargetPart = Settings.Part
        end
        if not Settings.Aimbot then
            fovCircle.Color = Settings.DayMode and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)
            lastTarget = nil
            lastTargetValid = false
            lastTargetPos = nil
            return
        end
        if tick() - lastUpdate < getUpdateInterval() then return end
        lastUpdate = tick()

        fovCircle.Color = Color3.fromRGB(255, 0, 0)
        
        local target = lastTarget
        if target and target.Parent and target.Position and lastTargetValid then
            local humanoid = target.Parent:FindFirstChild("Humanoid")
            if not humanoid or humanoid.Health <= 0 then
                lastTarget = nil
                lastTargetValid = false
                return
            end

            local screenPos, onScreen = Camera:WorldToViewportPoint(target.Position)
            if onScreen then
                local aimCenter = getAimingCenter()
                local distance = (Vector2.new(screenPos.X, screenPos.Y) - aimCenter).Magnitude
                local distanceToTarget = (target.Position - Camera.CFrame.Position).Magnitude
                if distance <= Settings.FOV and distanceToTarget < 1000 and target.Position.Magnitude < 10000 and target.Position.X == target.Position.X then
                    if not Settings.WallCheck or (tick() - lastRaycastTime >= raycastInterval and isVisible(target)) then
                        lastRaycastTime = tick()
                        lastTargetValid = true
                        lastTargetPos = target.Position
                    elseif tick() - lastRaycastTime < raycastInterval and lastTargetValid then
                    else
                        target = nil
                        lastTargetValid = false
                    end
                else
                    target = nil
                    lastTargetValid = false
                end
            else
                target = nil
                lastTargetValid = false
            end
        else
            target = getClosest()
            lastTarget = target
            lastTargetValid = false
            lastTargetPos = nil
            if target then
                lastRaycastTime = tick()
                lastTargetValid = isVisible(target)
                lastTargetPos = target and target.Position
            end
        end

        if not target or not target.Parent or not lastTargetPos then
            lastTarget = nil
            lastTargetValid = false
            return
        end

        local screenPos, onScreen = Camera:WorldToViewportPoint(lastTargetPos)
        if not onScreen then
            lastTarget = nil
            lastTargetValid = false
            return
        end

        local aimCenter = getAimingCenter()
        local targetScreenPos = Vector2.new(screenPos.X, screenPos.Y)
        local distance = (targetScreenPos - aimCenter).Magnitude
        if distance > Settings.FOV then
            lastTarget = nil
            lastTargetValid = false
            return
        end

        local distanceToTarget = (lastTargetPos - Camera.CFrame.Position).Magnitude
        if distanceToTarget < 1 or lastTargetPos.Magnitude > 10000 or lastTargetPos.X ~= lastTargetPos.X then
            lastTarget = nil
            lastTargetValid = false
            return
        end

        local degPerPixelX = Camera.FieldOfView / Camera.ViewportSize.Y
        local degPerPixelY = (Camera.FieldOfView * (Camera.ViewportSize.Y / Camera.ViewportSize.X)) / Camera.ViewportSize.Y

        local offsetXAngle = math.rad(Settings.AimOffsetX * degPerPixelX)
        local offsetYAngle = math.rad(Settings.AimOffsetY * degPerPixelY)
        offsetXAngle = math.clamp(offsetXAngle, -math.rad(45), math.rad(45))
        offsetYAngle = math.clamp(offsetYAngle, -math.rad(45), math.rad(45))
        local offsetCFrame = CFrame.Angles(0, -offsetXAngle, 0) * CFrame.Angles(-offsetYAngle, 0, 0)
        local adjustedCameraCFrame = Camera.CFrame * offsetCFrame

        local direction = (lastTargetPos - Camera.CFrame.Position).Unit
        local maxFOVAngle = math.rad(Settings.FOV / math.min(Camera.ViewportSize.X, Camera.ViewportSize.Y) * Camera.FieldOfView * 1.5)
        local currentLook = adjustedCameraCFrame.LookVector
        local angleToTarget = math.acos(currentLook:Dot(direction))
        if angleToTarget > maxFOVAngle then
            lastTarget = nil
            lastTargetValid = false
            return
        end

        local strength = Settings.AimStrength / 100
        local adjustedStrength = math.clamp((strength ^ 2) * 2, 0, 1)
        local smoothAlpha = math.clamp(1 - (strength * 0.9), 0.1, 0.9)
        local lockStrength = math.clamp(0.25 + adjustedStrength, 0.25, 1)

        local targetVelocity = target.Parent:FindFirstChild("HumanoidRootPart") and target.Parent.HumanoidRootPart.Velocity or Vector3.new(0, 0, 0)
        local predictionTime = Settings.AutoPredict and calculatePredictionTime(target, distanceToTarget) or 0
        local velocityAdjustment = targetVelocity * predictionTime
        local adjustedTargetPos = lastTargetPos + velocityAdjustment

        local predictedScreenPos, predictedOnScreen = Camera:WorldToViewportPoint(adjustedTargetPos)
        if predictedOnScreen then
        end

        local newCFrame = CFrame.new(Camera.CFrame.Position, adjustedTargetPos) * offsetCFrame:Inverse()
        if adjustedStrength > 0.8 then
            Camera.CFrame = newCFrame
        else
            Camera.CFrame = Camera.CFrame:Lerp(newCFrame, smoothAlpha * lockStrength)
        end
    end)
